Javascript (for customer support backend):

const express = require('express');
const bodyParser = require('body-parser');
const app = express();

// Middleware to parse form data
app.use(bodyParser.urlencoded({ extended: true }));

// Handle form submission
app.post('/action_page', (req, res) => {
    const { firstname, lastname, country, subject } = req.body;
    // Process the form data (e.g., save to a database, send an email, etc.)
    // You can customize this part based on your requirements
    console.log('Form submitted:', { firstname, lastname, country, subject });
    res.send('Thanks for contacting us!');
});

// Start the server
const PORT = process.env.PORT || 3000;
app.listen(PORT, () => {
    console.log(`Server running on port ${PORT}`);
});
